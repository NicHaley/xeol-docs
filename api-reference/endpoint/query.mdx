---
title: "Get Entity Info"
api: "POST https://eol.xeol.com/api/v1/query"
description: "This endpoint gets an entity's support information"
---

### Body

<ParamField body="identifiers" type="string[]" required="true">
  List of entity identifiers to get support information on

  <Expandable title="identifier">
    <ResponseField name="type" type="string">
      Identifier type for this entity
      
      Can be either `cpe` `purl` `custom`
    </ResponseField>

    <ResponseField name="identifier" type="string">
      Identifier string for the entity
      
      If `custom`, just use the entity's name like this: `Red Hat Enterprise Linux`
    </ResponseField>

    <ResponseField name="version" type="string">
      Version for this entity, supports wildcard like this `9.*`

      Only use if type is `custom` as versions are included in `cpe` and `purl`
    </ResponseField>
  </Expandable>
</ParamField>

### Response

<ResponseField name="results" type="object[]">
  List of entity with their listed supports

  <Expandable title="result">
    <ResponseField name="identifier" type="string">
      Identifier of the matched entity

      Can be either `cpe` `purl`
    </ResponseField>

    <ResponseField name="name" type="string">
      Name of the matched entity
    </ResponseField>

    <ResponseField name="version" type="string">
      Release cycle of the matched entity
    </ResponseField>

    <ResponseField name="type" type="string">
      Type of the matched entity
      
      Can be either `software` `package` `os` `product`
    </ResponseField>

    <ResponseField name="released" type="datetime">
      Release date of the matched entity
    </ResponseField>

    <ResponseField name="lts" type="boolean">
      Whether the matched entity has `Long Term Support`
    </ResponseField>

    <ResponseField name="support" type="object[]">
      All availabe support including `Extended Lifecycle Support` for matched entity
      
      This will always have at least **one** item.
      <Expandable title="support">
        <ResponseField name="id" type="id">
          Support ID 
        </ResponseField>

        <ResponseField name="name" type="string">
          Official support name
        </ResponseField>

        <ResponseField name="eol" type="datetime">
          End-of-life date
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
  ```bash curl
  curl --location --request POST 'https://eol.xeol.com/api/v1/query' \
  --header 'Content-Type: application/json' \
  --header 'Authorization: <XEOL_API_KEY>' \
  --data-raw '{
    "identifiers": [
      {
        "type": "custom",
        "identifier": "Red Hat Enterprise Linux",
        "version": "6"
      }
    ]
  }'
  ```
</RequestExample>
