---
title: "Check Software EOL"
api: "POST https://eol.xeol.com/api/v1/check"
description: "This endpoint checks if the provided software identifier(s) are end-of-life"
---

### Body

<ParamField body="identifiers" type="object[]" required="true">
  list of software identifiers to check for end-of-life status

  <Expandable title="identifier">
    <ResponseField name="type" type="string">
      Identifier type for this software.
      
      Can be either `cpe` `purl` `custom`
    </ResponseField>

    <ResponseField name="identifier" type="string">
      Unique identifier string for the software.
      
      If `custom`, just use the software's name like this: `Red Hat Enterprise Linux`
    </ResponseField>

    <ResponseField name="version" type="string">
      Version for this software
    </ResponseField>

    <ResponseField name="support" type="UUID">
      UUID for this software's support level including `Extended Lifecycle Support`

      If left out then it will default to `Standard Support`
    </ResponseField>
  </Expandable>
</ParamField>

### Response

<ResponseField name="results" type="object[]">
  List of matched artifacts with `eol` status

  <Expandable title="result">
    <ResponseField name="identifier" type="string">
      Software identifier string for the matched artifact
    </ResponseField>

    <ResponseField name="version" type="string">
      Software version for the matched artifact
    </ResponseField>

    <ResponseField name="eol" type="boolean">
      Whether the matched artifact is end-of-life.
      
      `null` if an artifact could not be matched.
    </ResponseField>

    <ResponseField name="artifact" type="object">
      Matched artifact info. 
      
      `null` if an artifact could not be matched.

      <Expandable title="artifact">
        <ResponseField name="name" type="string">
          Name of the matched artifact
        </ResponseField>

        <ResponseField name="version" type="string">
          Release cycle of the matched artifact
        </ResponseField>

        <ResponseField name="type" type="string">
          Type of the matched artifact, either `software` or `package` or `os` or `product`
        </ResponseField>

        <ResponseField name="released" type="datetime">
          Release date of the matched artifact
        </ResponseField>

        <ResponseField name="lts" type="boolean">
          Whether the matched artifact has `Long Term Support`
        </ResponseField>

        <ResponseField name="support" type="object[]">
          All availabe support including `Extended Lifecycle Support` for matched artifact
          
          This will always have at least **one** item.
          <Expandable title="support">
            <ResponseField name="id" type="UUID">
              Unique support ID 
            </ResponseField>

            <ResponseField name="name" type="string">
              Official support name
            </ResponseField>

            <ResponseField name="eol" type="datetime">
              End-of-life date
            </ResponseField>
          </Expandable>
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
  ```bash curl
  curl --location --request POST 'https://eol.xeol.com/api/v1/check' \
  --header 'Content-Type: application/json' \
  --header 'Authorization: <XEOL_API_KEY>' \
  --data-raw '{
      "identifiers": [
        {
          "type": "cpe",
          "identifier": "cpe:/o:alpinelinux:alpine_linux",
          "version": "3.14",
        },
        {
          "type": "purl",
          "identifier": "pkg:python/python@3.8",
          "version": "3.8",
          "support": "41a79382-9a93-4bec-a995-56fd7fdff9ff"
        }
      ]
  }'
  ```
</RequestExample>
